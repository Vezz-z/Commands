# 8

$ sudo ansible-galaxy init prerequisites
$ sudo nano prerequisites/tasks/main.yml

---
# tasks file for prerequisites 
- name: Install git 
  apt: 
    name: git 
    state: present 
    update_cache: yes

$ sudo ansible-galaxy init PostgreSQL
$ sudo nano PostgreSQL/tasks/main.yml

---
# tasks file for PostgreSQL 
- name: Install PostgreSQL 
  apt: 
    name: postgresql 
    state: present 
    update_cache: yes 
- name: Start PostgreSQL service 
  systemd: 
    name: postgresql 
    state: started 
    enabled: yes 

$ sudo ansible-galaxy init apache
$ sudo nano apache/tasks/main.yml

---
# tasks file for apache 
- name: install Apache web server 
  apt: 
    name: apache2 
    state: present 
    update_cache: yes

$ sudo nano stack.yml

---
- hosts: localhost 
  become: true 
  roles: 
        - prerequisites 
        - PostgreSQL 
        - apache

$ sudo ansible-playbook stack.yml
